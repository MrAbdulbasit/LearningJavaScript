Logical Operators


        Operator	Description
        >	        greater than
        <	        less than
        &&              And Sign
        ||              Or Sign
        !=	        not equal
        !==     	not equal value or not equal type
        ==	        equal to
        ===	        equal value and equal type
        >=	        greater than or equal to
        <=	        less than or equal to
    
    var a = 12;
    var b = a<13;
    var c = a>50;
    var d = b && c;
    var e = a<13 || a>50
    var f = a != b
    var g = a !== "12";
    var h = a === "12";
    var i = 50 >= 50;
    var j = 50 == 50;
    var k = 50 <= 50;
